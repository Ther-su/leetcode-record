## 空间压缩dp
```js
var rob = function(nums) {
  let n=nums.length;
  if(n==0) return 0;
  let dp0 = 0,dp1=nums[0]
  for(let i=1;i<n;i++) {
      let tdp0 = Math.max(dp0,dp1)
      let tdp1 = dp0+nums[i]
      dp0 = tdp0;
      dp1 = tdp1
  }
  return Math.max(dp0,dp1)
};
```
### 分析
dp0表示当前房屋不偷，dp0表示偷当前房屋

### 同样思路
```js
function solution2(nums) {
  let a = 0,b = 0;
  for(let i=0;i<nums.length;i++) {
      let c = Math.max(b,a+nums[i])
      a=b;
      b=c;
  }
  return b;
}
```
