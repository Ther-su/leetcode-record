/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var getKthFromEnd = function(head, k) {
    let fast=head,low=head;
    for(let i=0;i<k;i++){
        if(fast==null) return null;
        fast=fast.next;
    }
    while(fast!=null){
        fast=fast.next;
        low=low.next;
    }
    return low;
};
