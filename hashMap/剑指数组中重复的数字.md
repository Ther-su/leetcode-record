var findRepeatNumber = function(nums) {
    let temp;
    for(let i=0;i<nums.length;i++) {
        while(nums[i]!=i) {
            if (nums[i]==nums[nums[i]]) {
                return nums[i]
            }
            temp=nums[i]
            nums[i]=nums[temp]
            nums[temp]=temp;
        }
    }
    return -1;
};
